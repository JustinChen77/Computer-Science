import turtle
window=turtle.Screen()
t=turtle.Turtle()
t.pensize(5)

for Color in ["green"]:
    t.color(Color)
    t.forward(150)
    t.left(90)

for Color in ["blue"]:
    t.color(Color)
    t.forward(50)
    t.left(90)
    t.forward(50)
    t.right(90)
    t.forward(50)
    t.right(90)
    t.forward(50)
    t.left(90)
    t.forward(50)
    t.left(90)

for Color in ["cyan"]:
    t.color(Color)
    t.forward(150)
    t.left(90)

for Color in ["red"]:
    t.color(Color)
    t.forward(50)
    t.left(90)
    t.forward(50)
    t.right(90)
    t.forward(50)
    t.right(90)
    t.forward(50)
    t.left(90)
    t.forward(50)
